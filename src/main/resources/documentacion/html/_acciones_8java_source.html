246 public static byte[] CapturarFoto(ImageView imageView, BooleanProperty estadoCamara,
247 AtomicReference<Webcam> selWebCam, AtomicReference<BufferedImage> bufferedImage,
248 ObjectProperty<Image> imageProperty) {
249
250 if (Webcam.getWebcams().isEmpty()) {
251 Alert alert = new Alert(Alert.AlertType.INFORMATION, "No hay cámaras web disponibles", ButtonType.OK);
252 alert.showAndWait();
253 System.out.println("No hay cámaras");
254 return null;
255 }
256
257 if (!estadoCamara.get()) {
258 imageView.imageProperty().unbind();
259
260 Task<Void> webCamInitializer = new Task<>() {
261 @Override
262 protected Void call() throws Exception {
263 if (selWebCam.get() == null || !selWebCam.get().isOpen()) {
264 selWebCam.set(Webcam.getWebcams().get(0));
265 selWebCam.get().open();
266 }
267 estadoCamara.set(true);
268
269 Task<Void> task = new Task<>() {
270 @Override
271 protected Void call() throws Exception {
272 while (estadoCamara.get()) {
273 try {
274 BufferedImage currentImage = selWebCam.get().getImage();
275 bufferedImage.set(currentImage);
276
277 if (bufferedImage.get() != null) {
278 Platform.runLater(() -> {
279 final Image mainImage = SwingFXUtils.toFXImage(bufferedImage.get(), null);
280 imageProperty.set(mainImage);
281 });
282 bufferedImage.get().flush();
283 }
284 } catch (Exception e) {
285 e.printStackTrace();
286 }
287 }
288 return null;
289 }
290 };
291
292 Thread th = new Thread(task);
293 th.setDaemon(true);
294 th.start();
295 imageView.imageProperty().bind(imageProperty);
296 return null;
297 }
298 };
299
300 Thread webcamThread = new Thread(webCamInitializer);
301 webcamThread.setDaemon(true);
302 webcamThread.start();
303 return null;
304 } else {
305 estadoCamara.set(false);
306 if (selWebCam.get() != null) {
307 selWebCam.get().close();
308 }
309
310 // Convertir BufferedImage a byte[]
311 if (bufferedImage.get() != null) {
312 try (ByteArrayOutputStream baos = new ByteArrayOutputStream()) {
313 ImageIO.write(bufferedImage.get(), "jpg", baos);
314 return baos.toByteArray();
315 } catch (IOException e) {
316 e.printStackTrace();
317 }
318 }
319 }
320 return null;
321 }

392 public static byte[] ImageToByte(Stage stage) {
393 // Abrir un diálogo para seleccionar el archivo de imagen
394 FileChooser fileChooser = new FileChooser();
395 fileChooser.setTitle("Seleccionar imagen");
396 fileChooser.getExtensionFilters().addAll(
397 new FileChooser.ExtensionFilter("Archivos de Imagen", "*.jpg", "*.png", "*.jpeg")
398 );
399 File selectedFile = fileChooser.showOpenDialog(stage);
400
401 // Si el usuario no selecciona ningún archivo, devolver un arreglo de bytes vacío
402 if (selectedFile == null) {
403 return new byte[0];
404 }
405
406 // Convertir el archivo seleccionado a un arreglo de bytes
407 try (FileInputStream fis = new FileInputStream(selectedFile); ByteArrayOutputStream baos = new ByteArrayOutputStream()) {
408
409 byte[] buffer = new byte[1024];
410 int bytesRead;
411 while ((bytesRead = fis.read(buffer)) != -1) {
412 baos.write(buffer, 0, bytesRead);
413 }
414 return baos.toByteArray();
415
416 } catch (Exception e) {
417 System.out.println(e.getMessage());
418 return new byte[0];
419 }
420 }
