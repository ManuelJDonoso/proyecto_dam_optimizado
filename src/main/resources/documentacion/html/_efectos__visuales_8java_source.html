42 public static void closeEffectReducion(Node node) {
43 final Stage stage = (Stage) node.getScene().getWindow();
44 ScaleTransition st = new ScaleTransition(Duration.millis(500), node);
45 st.setToX(0);
46 st.setToY(0);
47 st.setToZ(0);
48 st.play();
49 st.setOnFinished(new EventHandler<ActionEvent>() {
50 @Override
51 public void handle(ActionEvent t) {
52 stage.close();
53 }
54 });
55 }

98 public static void Avatar(ImageView iv, double radius) {
99
100 // Establecer el tamaño del ImageView
101 iv.setFitWidth(radius * 2);
102 iv.setFitHeight(radius * 2);
103 iv.setPreserveRatio(true);
104
105 // Obtener las dimensiones de la imagen original para calcular el centro
106 double imageWidth = iv.getImage().getWidth();
107 double imageHeight = iv.getImage().getHeight();
108
109 // Calcular las coordenadas del centro de la imagen para centrado en el clip circular
110 double centerX = (imageWidth - radius * 2) / 2;
111 double centerY = (imageHeight - radius * 2) / 2;
112
113 // Aplicar un viewport centrado en la imagen
114 iv.setViewport(new Rectangle2D(centerX, centerY, radius * 2, radius * 2));
115
116 // Crear y aplicar el círculo como clip para el ImageView
117 Circle clip = new Circle(radius, radius, radius);
118 iv.setClip(clip);
119 }

130 public static void modificarTamañoImagen(ImageView iv, double radius, double scale) {
131 // Establecer el tamaño del ImageView según el radio (sin cambiar el radio)
132 iv.setFitWidth(radius * 2);
133 iv.setFitHeight(radius * 2);
134 iv.setPreserveRatio(true);
135
136 // Obtener la imagen original
137 Image image = iv.getImage();
138
139 // Calcular el nuevo tamaño de la imagen aplicando el factor de escala
140 double newWidth = image.getWidth() * scale;
141 double newHeight = image.getHeight() * scale;
142
143 // Crear una nueva imagen escalada
144 Image scaledImage = new Image(image.getUrl(), newWidth, newHeight, true, true);
145
146 // Establecer la imagen escalada al ImageView
147 iv.setImage(scaledImage);
148
149 // Obtener las nuevas dimensiones de la imagen escalada
150 double imageWidth = iv.getImage().getWidth();
151 double imageHeight = iv.getImage().getHeight();
152
153 // Calcular las coordenadas del centro de la imagen escalada para centrado en el clip circular
154 double centerX = (imageWidth - radius * 2) / 2;
155 double centerY = (imageHeight - radius * 2) / 2;
156
157 // Aplicar un viewport centrado en la imagen escalada
158 iv.setViewport(new Rectangle2D(centerX, centerY, radius * 2, radius * 2));
159
160 // Crear y aplicar el círculo como clip para el ImageView
161 Circle clip = new Circle(radius, radius, radius);
162 iv.setClip(clip);

170 public static void setBackgroundImage(Pane pane, String imagePath) {
171 // Cargar la imagen desde la ruta especificada
172 Image backgroundImage = new Image(imagePath);
173
174 // Configurar el fondo con la imagen
175 BackgroundImage background = new BackgroundImage(
176 backgroundImage,
177 BackgroundRepeat.NO_REPEAT, // No repetir la imagen
178 BackgroundRepeat.NO_REPEAT, // No repetir la imagen
179 BackgroundPosition.CENTER, // Centrar la imagen
180 new BackgroundSize(
181 BackgroundSize.AUTO, // Ancho automático
182 BackgroundSize.AUTO, // Alto automático
183 false, // No usar ancho total del AnchorPane
184 false, // No usar alto total del AnchorPane
185 true, // Ajustar al tamaño del contenedor
186 true // Ajustar para cubrir todo el contenedor
187 )
188 );
189
190 // Establecer el fondo en el AnchorPane
191 pane.setBackground(new Background(background));
192 }
