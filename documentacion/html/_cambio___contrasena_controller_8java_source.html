42 @Override
43 public void initialize(URL url, ResourceBundle rb) {
44 // Deshabilitar el botón aceptar al inicio
45 btn_aceptar.setDisable(true);

47 // Listener para verificar las contraseñas en ambos campos
48 ChangeListener<String> passwordListener = new ChangeListener<String>() {
49 @Override
50 public void changed(ObservableValue<? extends String> observable, String oldValue, String newValue) {
51 verificarContrasenas();
52 }
53 };
54
55 // Agregar el listener a ambos campos de contraseña
56 pf_contrasena.textProperty().addListener(passwordListener);
57 pf_contrasena_rep.textProperty().addListener(passwordListener);

60 @FXML

65 @FXML
66 private void btn_aceptar(ActionEvent event) {
67 botonAceptarPulsado = true; // Actualizar el estado al pulsar el botón
68 Efectos_visuales.closeEffectReducion(root);
69 }
70
71 private void verificarContrasenas() {
72 // Habilitar el botón si ambas contraseñas coinciden y no están vacías
73 boolean sonIguales = !pf_contrasena.getText().isEmpty()
74 && pf_contrasena.getText().equals(pf_contrasena_rep.getText());
75 btn_aceptar.setDisable(!sonIguales);
76 }
77
78 // Método para consultar si se ha pulsado el botón aceptar
79 public boolean isBotonAceptarPulsado() {
80 Session.setPass(pf_contrasena.getText());
81 return botonAceptarPulsado;
82 }
83
84 // Método opcional para reiniciar el estado del botón aceptar si es necesario
85 public void resetBotonAceptar() {
