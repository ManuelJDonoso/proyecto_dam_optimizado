182 public static void RegistrarProductoContrato(Usuario user) {
183 String usuario = user.getUsuario(), nombre = user.getNombre(), apellidos = user.getApellidos(), email = user.getEmail();
184 // Definir las consultas SQL para insertar los registros en ambas tablas
185 String sqlRegistro = "INSERT INTO Registro_tabla (id, Nombre, Apellidos, Email) VALUES (1, ?, ?, ?)";
186 String sqlUsuario = "INSERT INTO usuario_tabla (Usuario, Nombre, Apellidos, Email,fk_tipo) VALUES (?, ?, ?, ?,'ADMINISTRADOR')";
187
188 // Conexión y ejecución de las consultas dentro de una transacción
189 try (Connection conn = conectarSqlite()) {
190 // Iniciar una transacción
191 conn.setAutoCommit(false);
192
193 try (PreparedStatement pstmtRegistro = conn.prepareStatement(sqlRegistro); PreparedStatement pstmtUsuario = conn.prepareStatement(sqlUsuario)) {
194
195 // Establecer los parámetros para la tabla Registro_tabla
196 pstmtRegistro.setString(1, nombre);
197 pstmtRegistro.setString(2, apellidos);
198 pstmtRegistro.setString(3, email);
199
200 // Ejecutar la inserción en Registro_tabla
201 pstmtRegistro.executeUpdate();
202
203 // Establecer los parámetros para la tabla usuario_tabla
204 pstmtUsuario.setString(1, usuario); // Usuario como parámetro
205 pstmtUsuario.setString(2, nombre);
206 pstmtUsuario.setString(3, apellidos);
207 pstmtUsuario.setString(4, email);
208
209 // Ejecutar la inserción en usuario_tabla
210 pstmtUsuario.executeUpdate();
211
212 // Confirmar la transacción
213 conn.commit();
214 System.out.println("Registros insertados exitosamente en Registro_tabla y usuario_tabla.");
215
216 } catch (SQLException e) {
217 // En caso de error, revertir la transacción
218 conn.rollback();
219 System.out.println("Error al insertar los registros: " + e.getMessage());
220 }
221
222 } catch (SQLException e) {
223 System.out.println("Error al conectar con la base de datos: " + e.getMessage());
